To Build the Tree:

First run:

wget ftp://greengenes.microbio.me/greengenes_release/gg_13_5/gg_13_8_otus.tar.gz
tar -xvzf gg_13_8_otus.tar.gz

From that folder, you are looking for trees/97_otus.tree specifically

Then run:

wget -c \
  -O "sepp-refs-gg-13-8.qza" \
  "https://data.qiime2.org/2024.2/common/sepp-refs-gg-13-8.qza"

to get the SEPP reference data


qiime tools import \
  --input-path 97_otus.tree \
  --output-path 97_otus.tree.qza \
  --type 'Phylogeny[Unrooted]'

qiime phylogeny midpoint-root \
  --i-tree 97_otus.tree.qza \
  --o-rooted-tree 97_otus_rooted.qza

qiime fragment-insertion sepp \
  --i-representative-sequences rep-seqs.qza \
  --i-reference-database sepp-refs-gg-13-8.qza \
  --o-tree insertion-tree.qza \
  --o-placements insertion-placements.qza

qiime fragment-insertion sepp \
  --i-representative-sequences rep_seqs.qza \
  --i-reference-database sepp-refs-gg-13-8.qza \
  --o-tree insertion-tree.qza \
  --o-placements insertion-placements.qza \
  --p-threads 4 \
  --p-alignment-subset-size 500 \
  --p-placement-subset-size 1000 \
  --verbose
